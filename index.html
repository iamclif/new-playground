var dogs = ['mutt', 'boxer', 'pug'];
var animals = ['whale', 'snake', ...cats, ...dogs, 'coyote', 'bear'];

var container = document.getElementById("existingDiv");
for (var item in animals) {
    var paragraph = document.createElement("p");
    paragraph.innerHTML = animals[item];
    container.appendChild(paragraph);
}

Arrow Functions and "this" scope

ECMAScript 5



var person = {
  first: 'Doug',
  actions: ['bike', 'hike', 'run'],
  printActions: function(){
    var _this = this;
    this.actions.forEach(function(actions){
        var str = _this.first + " likes to " + actions;
        console.log(str)
    });
  }
}; 
or
var person = {
  first: 'Doug',
  actions: ['bike', 'hike', 'run'],
  printActions: function(){
    this.actions.forEach(function(actions){
        var str = _this.first + " likes to " + actions;
        console.log(str)
    }.bind(this));
  }
};
person.printActions();

ECMAScript 6



var person = {
  first: 'Doug',
  actions: ['bike', 'hike', 'run'],
  printActions() {
    this.actions.forEach(actions => {
        var str = this.first + " likes to " + actions;
        console.log(str)
    });
  }
};
person.printActions();
