Arrow Functions and "this" scope

ECMAScript 5



var person = {
  first: 'Doug',
  actions: ['bike', 'hike', 'run'],
  printActions: function(){
    var _this = this;
    this.actions.forEach(function(actions){
        var str = _this.first + " likes to " + actions;
        console.log(str)
    });
  }
}; 
or
var person = {
  first: 'Doug',
  actions: ['bike', 'hike', 'run'],
  printActions: function(){
    this.actions.forEach(function(actions){
        var str = _this.first + " likes to " + actions;
        console.log(str)
    }.bind(this));
  }
};
person.printActions();

Generators

ECMAScript 6


function* eachItem(arr) {
  for(var i=0; i < arr.length; i++) {
    yield arr[i];
  }
}

var letters = eachItem(['a', 'b', 'c', 'd', 'e', 'f', 'g']);

var abcs = setInterval(function(){
  var letter = letters.next();
  var container = document.getElementById('abcs');
  var paragraph = document.createElement('p');
  if(letter.done) {
    clearInterval(abcs);
    paragraph.innerHTML = 'Now I Know My ABCs';
    container.appendChild(paragraph);
  } else {
    var foo = letter.value;
    paragraph.innerHTML = foo;
    container.appendChild(paragraph);
  }
},500);
